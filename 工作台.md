// 色调(平均色会互相抵消成灰)。失败必须打日志——静默失败跟"取色

      const c = document.createElement("canvas");
      c.width = 32; c.height = 32;
      const ctx = c.getContext("2d");
      ctx.drawImage(img, 0, 0, 32, 32);
      const d = ctx.getImageData(0, 0, 32, 32).data;
      const bins = Array.from({ length: 12 }, () => ({ r: 0, g: 0, b: 0, n: 0 }));
      for (let i = 0; i < d.length; i += 4) {
        const r = d[i], g = d[i + 1], b = d[i + 2];

        if (mx - mn < 24) continue; // 近灰像素没有可用色相,不投票

        const bin = bins[Math.floor(h * 2) % 12];
        bin.r += r; bin.g += g; bin.b += b; bin.n++;
      }
      const top = bins.reduce((a, x) => (x.n > a.n ? x : a));
      if (top.n) {

// —— 随机重逢:随机抽一篇旧笔记,可换一篇。排除 notes/问答/(那是

// 摘要能放几行按裁完的实际剩余高度算,写进 line-clamp——固定行数一旦
// 超过可用高度,超高的文字块会被裁切得看不到开头。窄屏两卡上下堆叠,
// 清掉限制按内容自然展开(CSS 里有窄屏 8 行的兜底)
const syncReunionHeight = () => {
  const painting = wbRoot?.querySelector(".wb-daily-painting");
  const text = reunionEl.querySelector(".wb-reunion-excerpt-text");
  if (!painting) return;
  if (window.innerWidth <= 520) {
    reunionEl.style.maxHeight = "";
    if (text) text.style.webkitLineClamp = "";
    return;
  }
  const h = painting.getBoundingClientRect().height;
  if (h <= 0) return;
  reunionEl.style.maxHeight = h + "px";
  const box = reunionEl.querySelector(".wb-reunion-excerpt");
  if (box && text) {
    const lh = parseFloat(getComputedStyle(text).lineHeight) || 20;
    text.style.webkitLineClamp = String(Math.max(1, Math.floor(box.clientHeight / lh)));
  }
};

const renderReunion = async (container) => {
  container.innerHTML = "";
  const arr = dv.pages('"notes"').where(pg => !pg.file.folder.startsWith("notes/问答")).array();
  if (!arr.length) return;
  const pick = arr[Math.floor(Math.random() * arr.length)];
  const head = container.createEl("div", { cls: "wb-reunion-head" });
  head.createEl("span", { cls: "wb-reunion-tag", text: "🎲 随机重逢" });
  const reroll = head.createEl("button", { cls: "wb-reunion-reroll" });
  const rerollGlyph = reroll.createSpan({ cls: "wb-reroll-glyph", text: "↻" });
  container.createEl("div", { cls: "wb-reunion-title", text: pick.file.name });
  const excerptEl = container.createEl("div", { cls: "wb-reunion-excerpt" });
  const excerptTextEl = excerptEl.createEl("div", { cls: "wb-reunion-excerpt-text", text: "…" });
  container.onclick = () => open(pick.file.path);
  // 换一篇的旋转用一次性 Web Animations 动画(手机没有真 hover),
  // 必须等 finished 再重建卡片,否则动画还没画出一帧节点就被销毁了
  reroll.onclick = async (e) => {
    e.stopPropagation();
    if (reroll.disabled) return;
    reroll.disabled = true;
    const anim = rerollGlyph.animate(
      [{ transform: "rotate(0deg)" }, { transform: "rotate(360deg)" }],
      { duration: 450, easing: "ease" }
    );
    try { await anim.finished; } catch (_) { /* 动画被打断也照样换 */ }
    // 清掉可能残留的内联 transform,免得压死 CSS 的 hover 旋转
    rerollGlyph.style.removeProperty("transform");
    renderReunion(container);
  };
  const raw = await dv.io.load(pick.file.path).catch(() => "");
  const body = (raw || "")
    .replace(/^---[\s\S]*?---/, "")
    .replace(/\[\[([^\]|]*)\|?[^\]]*\]\]/g, "$1")
    .replace(/[#>*`_]/g, "")
    .trim();
  // 1500 字是宽松上限,给够料让 line-clamp 按实际高度裁
  const snippet = body.slice(0, 1500).replace(/\s+/g, " ").trim();
  excerptTextEl.setText(snippet + (body.length > 1500 ? "…" : ""));
  syncReunionHeight();
};

// 两个 dataviewjs 块渲染顺序不保证,轮询等画作卡 DOM 就绪再挂观察。
// 观察者/监听器全部注册进 dv.component,本块重渲染时自动清理——
// 否则每次重渲染都叠一个观察者,旧的拿着过期高度反复写回,互相打架
let cancelled = false;
dv.component.register(() => { cancelled = true; });
const setupLayoutSync = (triesLeft = 34) => {
  if (cancelled) return;
  const paintingEl = wbRoot?.querySelector(".wb-daily-painting");
  if (!paintingEl) {
    if (triesLeft > 0) setTimeout(() => setupLayoutSync(triesLeft - 1), 150);
    else console.warn("[工作台] 等了 5 秒没找到画作卡片 DOM,放弃高度同步——可能画作代码块报错了");
    return;
  }
  syncReunionHeight();
  if (window.ResizeObserver) {
    const ro = new ResizeObserver(syncReunionHeight);
    ro.observe(paintingEl);
    dv.component.register(() => ro.disconnect());
  }
  dv.component.registerDomEvent(window, "resize", syncReunionHeight);
};
setupLayoutSync();

> **问题：** 水缸的过滤器靠"消耗感/紧绷感"这种身体信号,但有没有一种人初期不触发紧绷感、缸里待久了才开始放毒?过滤器怎么防这种慢渗透?

> **你的回答：** 肯定有这种人,只要发现他有问题就跑。
>
> **我的看法：** 这种人确实存在,而且是最危险的一类——初期给的是新鲜感/认同感这类正向刺激,身体的即时警报系统是为"威胁"设计的,不是为"缓慢透支"设计的。真正能防慢渗透的是换一把尺子:定期回头问"和这个人相处一段时间后,我更像自己了还是更不像自己了",拿跨时间的基线对比,而不是单次互动的即时读数。

> [!clash]- 🥊 碰撞 · 问题2 回顾
> **问题：** 控制二分法会不会被滥用成回避的借口——把其实可控(只是难)的事划进"不可控"提前放弃?判据是什么?
>
> **你的回答：** 只要不是放弃了就彻底松一口气,那就是逃避;真正的情况是客观上觉得自己大概率做不到才不做。
>
> **我的看法：** 判据是问自己"如果真的投入努力、技巧或时间,结果发生的概率会不会变"。真不可控的事,不管你多努力概率纹丝不动;假的"不可控"其实是"太难了",概率会随投入变化,只是变得慢、变得不舒服。还有一个更快的识别信号:说出"这不是我能控制的"之后,如果立刻如释重负、然后就此打住没有任何后续动作——大概率是借口。
>
> **碰撞：** 我们用的其实是同一枚硬币的两面,但你这枚判据站不住脚。"事后有没有彻底放松"这个信号在真逃避和真理性放弃两种情况下都会出现——放弃一件真正拼过全力也做不到的事,同样会带来彻底的如释重负,因为长期悬着的焦虑终于卸下了;而一个正在逃避的人,只要给自己找到了足够体面的借口,同样会彻底放松,不会留下任何"没放松"的破绽让你抓。用"是否彻底放松"去分辨两者,等于用一个两边都会触发的信号去做判别,基本等价于没有判据。真正能分辨的还是那把尺子——投入是否真的改变过结果概率,这是可以回头核查行动记录的客观事实,不依赖你事后怎么描述自己的情绪。
