const setupLayoutSync = (triesLeft = 34) => {

  if (!paintingEl) {

    else console.warn("[工作台] 等了 5 秒还是没找到画作卡片的 DOM,放弃高度同步——可能画作代码块本身报错了");
    return;
  }
  syncLayout();
  // 2026-07-07 修:之前这个 ResizeObserver 和 resize 监听器从没清理过,
  // 跟上面 tick() 定时器踩过的是同一个坑(见文件顶部那段注释)——这个
  // dataviewjs 块只要重新渲染一次(切页面再回来、或文件被外部改动触发
  // Dataview 重渲),就会在新一份 paintingEl 上再挂一对新的观察者,旧的
  // 那对还在盯着可能已经从 DOM 摘掉的旧画作卡,两份回调谁后触发就由谁
  // 把 reunionEl.style.maxHeight 写成谁量到的值,新旧数值互相打架,表现
  // 就是"随机重逢的高度跟画作对不上"。改成用 dv.component.register 把
  // disconnect/removeEventListener 绑定到这次渲染的生命周期上,组件卸载
  // (离开/重渲染这个块)时自动清掉,不再累积。
  if (window.ResizeObserver) {
    const ro = new ResizeObserver(syncLayout);
    ro.observe(paintingEl);
    dv.component.register(() => ro.disconnect());
  }
  window.addEventListener("resize", syncLayout);
  dv.component.register(() => window.removeEventListener("resize", syncLayout));
};
