      const c = document.createElement("canvas");

      const ctx = c.getContext("2d");
      ctx.drawImage(img, 0, 0, 32, 32);
      const d = ctx.getImageData(0, 0, 32, 32).data;
      const bins = Array.from({ length: 12 }, () => ({ r: 0, g: 0, b: 0, n: 0 }));
      for (let i = 0; i < d.length; i += 4) {

        if (mx - mn < 24) continue; // 近灰的像素没有可用色相,不参与投票

      // 头(canvas 被"跨域污染",getImageData 直接抛 SecurityError)
      console.warn("[工作台] 画作取色失败,沿用兜底色 --wb-accent:", e);
    }
  };
  img.onerror = (e) => {

    // 表现上跟"取色成功但算出兜底同款颜色"一模一样,不打日志根本
    // 分不清是这里失败还是上面 try/catch 里失败
    console.warn("[工作台] 画作图片加载失败,取色跳过,沿用兜底色 --wb-accent。图片地址:", p["painting-url"], e);
  };
  img.src = p["painting-url"];
}

const syncReunionHeight = () => {

  if (!painting) return;
  if (window.innerWidth <= 520) {
    reunionEl.style.maxHeight = "";
    return;
  }
  const h = painting.getBoundingClientRect().height;
  if (h > 0) reunionEl.style.maxHeight = h + "px";
};
if (window.ResizeObserver) {
  const paintingEl = document.querySelector(".workbench .wb-daily-painting");
  if (paintingEl) new ResizeObserver(syncReunionHeight).observe(paintingEl);
}
window.addEventListener("resize", syncReunionHeight);

> 关于 [[concepts/胰岛素抵抗|胰岛素抵抗]]:两条原始笔记都只是分散提到它,还没有一个可操作的自测/确诊方法。最常用的临床估算指标是 **HOMA-IR**(稳态模型评估-胰岛素抵抗指数):公式为「空腹血糖(mmol/L)× 空腹胰岛素(μIU/mL) ÷ 22.5」,数值越高代表胰岛素抵抗越明显,通常 >2.5~3 被视为存在胰岛素抵抗的参考阈值(不同实验室/人群切点略有差异)。[事实,未联网核查]
