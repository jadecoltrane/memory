// —— 笔记库:按分类的卡片墙(实时) ——
const pages = dv.pages('"notes"').array();
const groups = new Map();
for (const p of pages) {
  const seg = p.file.folder.split("/");
  const cat = seg.length > 1 ? seg[1] : "未分类";
  if (!groups.has(cat)) groups.set(cat, []);
  groups.get(cat).push(p);
}
for (const [cat, list] of [...groups.entries()].sort((a, b) => b[1].length - a[1].length)) {
  list.sort((a, b) => (b.created?.ts ?? 0) - (a.created?.ts ?? 0));
  const sec = dv.container.createEl("div", { cls: "wb-lib-group" });
  const head = sec.createEl("div", { cls: "wb-lib-head" });
  head.createEl("span", { cls: "wb-lib-cat", text: cat });
  head.createEl("span", { cls: "wb-lib-count", text: String(list.length) });
  const grid = sec.createEl("div", { cls: "wb-lib-grid" });
  for (const p of list) {
    const card = grid.createEl("a", { cls: "wb-lib-card" });
    card.onclick = () => app.workspace.openLinkText(p.file.path, "/", false);
    card.createEl("div", { cls: "wb-lib-title", text: p.file.name });
    const meta = card.createEl("div", { cls: "wb-lib-meta" });
